77 class UndoManager {
78   private var undoStack: List[Command] = Nil
79   private var redoStack: List[Command] = Nil
80 
81   def doStep(command: Command): Unit = {
82     undoStack = command :: undoStack
83     command.doStep()
84     redoStack = Nil
85   }
86 
87   def undoStep(): Unit = {
88     undoStack match {
89       case Nil => // no-op
90       case head :: stack =>
91         head.undoStep()
92         undoStack = stack
93         redoStack = head :: redoStack
94     }
95   }
96 
97   def redoStep(): Unit = {
98     redoStack match {
99       case Nil => // no-op
100       case head :: stack =>
101         head.redoStep()
102         redoStack = stack
103         undoStack = head :: undoStack
104     }
105   }
106 }
